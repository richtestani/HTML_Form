What is this?.

HTML Form helps build HTML form elements within your PHP application.

Simple Example

include 'Form.php';

//
$form = new Form('/form_process.php', 'post');

//insert new input
$form->insertElement('text', 'username', '', array('class'=>'form-element', 'placeholder'=>'username'));
$form->setLabel('Username', 'username');

$form->insertElement('text', 'password', '', array('class'=>'form-element', 'placeholder'=>'password'));
$form->setLabel('Password', 'password');

//get the resulting HTML
$finalForm = $form->get();
